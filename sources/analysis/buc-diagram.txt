@startuml
title BUC-диаграмма\nБизнес-прецеденты интернет-магазина зоотоваров «Лапки»

' Определение актёров
actor "Гость" as Guest
actor "Авторизованный\nпользователь" as User
actor "Администратор" as Admin

' Бизнес-прецеденты для Гостя
rectangle "Бизнес-процессы платформы «Лапки»" {
  
  ' Гость
  usecase "Ознакомление с каталогом\n(просмотр, поиск, фильтрация)" as UC1
  usecase "Просмотр карточки товара\nи отзывов" as UC2
  usecase "Регистрация / Вход\nв систему" as UC3
  
  ' Авторизованный пользователь
  usecase "Управление личным профилем\n(аватар, контактные данные)" as UC4
  usecase "Управление корзиной\n(добавление, удаление, изменение)" as UC5
  usecase "Оформление заказа" as UC6
  usecase "Управление избранным" as UC7
  usecase "Оставление отзыва\nс рейтингом" as UC8
  usecase "Просмотр истории заказов" as UC9
  
  ' Администратор
  usecase "Управление каталогом\n(CRUD товаров, категорий,\nвидов животных)" as UC10
  usecase "Управление пользователями" as UC11
  usecase "Модерация отзывов" as UC12
  usecase "Просмотр статистики\nи отчётов" as UC13
}

' Связи актёров с прецедентами
Guest --> UC1
Guest --> UC2
Guest --> UC3

User --> UC4
User --> UC5
User --> UC6
User --> UC7
User --> UC8
User --> UC9

Admin --> UC10
Admin --> UC11
Admin --> UC12
Admin --> UC13

' Наследование (Администратор наследует всё, что есть у пользователя)
Admin --|> User
User --|> Guest

@enduml
